\frame{
  \frametitle{Características chave}

  \begin{itemize}
    \item \textbf{``\textit{Cloud Operating System}''}
      \begin{itemize}
        \item Diferentes tipos de hardware
        \vspace{.5em}
        \item Suporte a praticamente todos os \textit{hypervisors} comerciais\footnote{Inclusive LXC/LXD}
        \vspace{.5em}
        \item Integração completa com os principais serviços de \textit{IaaS}
        \vspace{.5em}
        \item Representação de topologias de rede complexas
        \vspace{.5em}
        \item Gerenciamento de sistemas de arquivos
      \end{itemize}
    \vspace{1em}
    \item \textbf{Lado negativo:} com grandes poderes, vêm grandes responsabilidades
      \begin{itemize}
        \item Instalação e gerenciamento mais complexos que outros concorrentes\footnote{Experimente o OpenNebula por exemplo ;)}
      \end{itemize}
  \end{itemize}
}
